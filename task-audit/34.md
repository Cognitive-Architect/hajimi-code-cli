片段A：TURN客户端核心（来自 turn-client.ts）：

```typescript
// RFC 5766 TURN客户端实现
export class TURNClient {
  async allocate(): Promise<RelayCandidate> {
    // 401挑战响应
    const response = await this.sendAllocateRequest();
    if (response.code === 401) {
      const nonce = response.attributes.nonce;
      const realm = response.attributes.realm;
      const integrity = this.calculateIntegrity(nonce, realm); // HMAC-SHA1
      return this.sendAllocateRequest(integrity); // 二次请求带认证
    }
    if (response.code === 403) throw new Error('TURN_FORBIDDEN'); // 不重试
    return this.parseRelayCandidate(response);
  }
  
  private calculateIntegrity(nonce: string, realm: string): string {
    const key = createHmac('sha1', this.password)
      .update(`${this.username}:${realm}:${this.password}`)
      .digest();
    return key.toString('base64');
  }
}
```

片段B：ICE候选优先级与降级（来自 ice-manager.ts）：

```typescript
enum CandidateType { HOST = 'host', SRFLX = 'srflx', RELAY = 'relay' }

const PRIORITY = {
  [CandidateType.HOST]: 126,    // LAN直连优先
  [CandidateType.SRFLX]: 100,   // WAN直连次之
  [CandidateType.RELAY]: 80     // TURN中继最后
};

export class ICEManager {
  private connectionState: ConnectionState = ConnectionState.LAN;
  
  async connect(peerId: string): Promise<Connection> {
    // 1. 尝试host候选（LAN）
    if (await this.tryHost(peerId)) return this.setState(ConnectionState.LAN);
    // 2. 尝试srflx候选（WAN直连）
    if (await this.trySrflx(peerId)) return this.setState(ConnectionState.DIRECT);
    // 3. 尝试relay候选（TURN）
    if (await this.tryRelay(peerId)) return this.setState(ConnectionState.RELAY);
    throw new Error('ICE_FAILED');
  }
}
```

片段C：Benchmark RSS内存测量（来自 p2p-sync-benchmark.js）：

```javascript
const v8 = require('v8');
const { performance } = require('perf_hooks');

class P2PSyncBenchmark {
  measureMemory() {
    const heapStats = v8.getHeapStatistics();
    const rss = process.memoryUsage().rss; // 关键：RSS非堆内存
    return {
      heapUsed: heapStats.used_heap_size,
      rss: rss, // 包含C++层内存（LevelDB等）
      external: heapStats.external_memory
    };
  }
  
  async runBenchmark(chunkCount) {
    const before = this.measureMemory();
    const start = performance.now();
    
    await this.syncChunks(chunkCount); // 实际同步
    
    const duration = performance.now() - start;
    const after = this.measureMemory();
    return {
      duration,
      maxRss: Math.max(before.rss, after.rss),
      tps: chunkCount / (duration / 1000),
      p95Latency: this.calculateP95() // 基于采样
    };
  }
}
```
