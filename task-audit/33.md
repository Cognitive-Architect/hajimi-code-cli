```typescript
merge(local: Chunk, remote: Chunk): Chunk {
  const mergedDoc = new Y.Doc();
  Y.applyUpdate(mergedDoc, local.crdtState);
  Y.applyUpdate(mergedDoc, remote.crdtState); // CRDT自动合并，无timestamp比较
  return this.docToChunk(mergedDoc);
}
```

片段B：LevelDB持久化核心（来自 p2p-queue-db.ts）：

```typescript
export class P2PQueueDB implements IQueueDb {
  private db: Level<string, string>;
  
  async saveQueue(queue: SyncOperation[]): Promise<void> {
    const batch = this.db.batch();
    for (const op of queue) {
      batch.put(op.id, JSON.stringify(op));
    }
    await batch.write(); // ACID原子写入
  }
  
  async getQueue(): Promise<SyncOperation[]> {
    const ops: SyncOperation[] = [];
    for await (const [key, value] of this.db.iterator()) {
      ops.push(JSON.parse(value));
    }
    return ops;
  }
}
```

片段C：最终整合版关键（来自 bidirectional-sync-final.ts，282行）：

```typescript
export class BidirectionalSyncFinal extends SyncEngine {
  private crdtEngine: ICrdtEngine;  // B-40/01产出
  private queueDb: IQueueDb;        // B-40/02产出
  
  // 启动时恢复队列
  private async restoreQueue(): Promise<void> {
    this.offlineQueue = await this.queueDb.getQueue();
  }
  
  // CRDT合并后持久化
  private async persistState(): Promise<void> {
    await this.queueDb.saveQueue(this.offlineQueue);
  }
}
```
